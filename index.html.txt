<script>
  const SCRIPT_URL = 'https://script.google.com/macros/s/AKfycbyk8RPMVDAEIkaFWPb8lUP_DazjUcKZn5Ah9whFuPTAsBKfDynloXMvfNQ1yNBjbPajoQ/exec'; //
  let currentRow = null;
  let currentData = null;

  async function submitRequest() {
    const link = document.getElementById('productLink').value.trim();
    const name = document.getElementById('productName').value.trim();
    const contact = document.getElementById('contact').value.trim();
    const country = document.getElementById('country').value;

    if (!link) {
      alert('상품 링크를 입력해주세요 · Введите ссылку на товар');
      return;
    }

    // 로딩 표시
    document.getElementById('formCard').style.display = 'none';
    document.getElementById('loader').style.display = 'block';

    // 임시 한국판매가 (운영자가 나중에 입력) - MVP용
    const tempKoreanPrice = 0;

    currentData = {
      productLink: link,
      productName: name || '상품명 확인 필요',
      contact: contact,
      country: country,
      koreanPrice: tempKoreanPrice,
      selectedStore: '확인 필요'
    };

    try {
      const res = await fetch(SCRIPT_URL, {
        method: 'POST',
        body: JSON.stringify(currentData)
      });
      const result = await res.json();

      document.getElementById('loader').style.display = 'none';

      if (result.success) {
        showResult(result);
      } else {
        throw new Error(result.error);
      }
    } catch(err) {
      document.getElementById('loader').style.display = 'none';
      document.getElementById('formCard').style.display = 'block';
      alert('오류가 발생했습니다. 다시 시도해주세요.');
    }
  }

  function showResult(result) {
    document.getElementById('resultProductName').textContent =
      currentData.productName;

    document.getElementById('r_koreanPrice').textContent =
      result.koreanPrice ? formatKRW(result.koreanPrice) : '확인 중...';
    document.getElementById('r_proxyFee').textContent =
      result.proxyFee ? formatKRW(result.proxyFee) : '확인 중...';
    document.getElementById('r_shippingFee').textContent =
      result.shippingFee ? formatKRW(result.shippingFee) : '확인 중...';
    document.getElementById('r_finalPrice').textContent =
      result.finalPrice ? formatKRW(result.finalPrice) : '확인 중...';

    document.getElementById('resultCard').style.display = 'block';
  }

  async function submitIntent(intent) {
    await fetch(SCRIPT_URL, {
      method: 'POST',
      body: JSON.stringify({
        ...currentData,
        intent: intent
      })
    });
    document.getElementById('resultCard').style.display = 'none';
    document.getElementById('successMsg').style.display = 'block';
  }

  function formatKRW(amount) {
    return amount.toLocaleString('ko-KR') + '원';
  }

  function resetForm() {
    document.getElementById('successMsg').style.display = 'none';
    document.getElementById('formCard').style.display = 'block';
    document.getElementById('productLink').value = '';
    document.getElementById('productName').value = '';
    document.getElementById('contact').value = '';
  }
</script>
